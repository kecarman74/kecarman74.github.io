void craps(){
    int win{0};
    int loss{0};
    array<int, 20> winFreq{};
    array<int, 20> loseFreq{};
    double avgRoll{0};
    for (size_t i{0}; i <= 20; i++){
        winFreq[i] = 0;
        loseFreq[i] = 0;
    }
    for (int games{1000}; games > 0; games--){
        enum class Status{CONTINUE, WON, LOST};
        Status gameStatus;
        unsigned int myPoint{0};
        unsigned int sumOfDice{rollDice()};
        switch (sumOfDice){
            case 7:
            case 11:
                gameStatus = Status::WON;
                break;
            case 2:
            case 3:
            case 12:
                gameStatus = Status::LOST;
                break;
            default:
                gameStatus = Status::CONTINUE;
                myPoint = sumOfDice;
                break;
        }
        int roll{1};
            while (gameStatus == Status::CONTINUE){
                sumOfDice = rollDice();
                roll += 1;
                if (sumOfDice == myPoint){
                    gameStatus = Status::WON;
                }
                else {
                    if (sumOfDice == 7){
                        gameStatus = Status::LOST;
                    }
                }
            }
            if (Status::WON == gameStatus){
                win += 1;
                if (roll >= 20){
                    roll = 20;
                }
                ++winFreq[roll];
            }
            else {
                loss += 1;
                if (roll >= 20){
                    roll = 20;
                }
                ++loseFreq[roll];
            }
        }
    cout << " Roll Number " << " Number of Wins " << " Number of Losses " << endl;
    for (size_t j{0}; j <= 20; j++){
            cout << setw(9) << j + 1 << setw(15) << winFreq[j] << setw(15) << loseFreq[j] << endl;
            avgRoll += (((j + 1) * winFreq[j]) + ((j + 1) * loseFreq[j])) ;
    }
    cout << "Total Wins: " << win << endl;
    cout << "Total Losses: " << loss << endl;
    cout << "Win Percentage: " << (win / 10) << "%" << endl;
    cout << "Avg Game Length: " << avgRoll / 1000 << " Rolls" << endl;
}

array<int, 20> minArr{7, 2, 3, 5, 6, 7, 2, 3, 5, 6, 7, 2, 3, 5, 6, 7, 2, 3, 5, 6};
int minArray(array<int, 20> arr, int start, int stop){
    int index{start + 1};
    if (start == stop){
        return arr[start];
    }
    else{
        int min{arr[start]};
        if(min > arr[index]){
            min = arr[index];
            minArray(arr, index, stop);
        }
        return min;
    }
}
