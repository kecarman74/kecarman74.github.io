DeckOfCards::DeckOfCards()
{
    //ctor
    currentCard = 0;
    initDeck();
    shuffle();
}

    void DeckOfCards::initDeck() {

        for (int i{0}; i < 4; i++) {

            for (int j{0}; j < 13; j++) {

                deck.push_back(Card(static_cast<Face>(j), static_cast<Suit>(i)));
            }
        }
    }

    void DeckOfCards::shuffle() {
        for (int i{0}; i < 10000; i++) {
            Card temp;
            int j{0};
            int k{0};
            j = rand() % 51;
            k = rand() % 51;
            temp = deck[j];
            deck[j] = deck[k];
            deck[k] = temp;
        }
    }

   Card DeckOfCards::dealCard() {

        int card{currentCard};
        currentCard++;

        return deck[card];
    }

    bool DeckOfCards::moreCards() const{
        bool moCards{true};
        if (currentCard == 51) {
            moCards = false;
        }
        return moCards;
    }
