#define setSize 101

IntegerSet::IntegerSet() {}

IntegerSet::IntegerSet(vector<int> newArr) : init{newArr}
{
    //ctor
    intArrToBoolArr();
}

void IntegerSet::intArrToBoolArr() {

     for (int i{0}; i < setSize; i++) {

        setOfInts[i] = false;
    }

    for (int i{0}; i < init.size(); i++) {

        setOfInts[init[i]] = true;
    }
}

 string IntegerSet::toString() const {
    string intSet;

    for (int i{0}; i < setSize; i++) {

        if (setOfInts[i] == true) {
            intSet += to_string(i);
            intSet += " ";
        }
    }
    return intSet;
 }

void IntegerSet::insertElement(int ins) {
    setOfInts[ins] = true;
}

void IntegerSet::deleteElement(int del) {
    setOfInts[del] = false;
}

IntegerSet IntegerSet::intersectionOfSets(const IntegerSet& other) const{
    vector<int> newVect;

    for (int i{0}; i < setSize; i++) {

        if (setOfInts[i] && other.setOfInts[i]) {
            newVect.push_back(i);
        }
    }
    return IntegerSet(newVect);
}

IntegerSet IntegerSet::unionOfSets(const IntegerSet& other) const{
    vector<int> newVect;

    for (int i{0}; i < setSize; i++) {

        if (setOfInts[i] || other.setOfInts[i]) {
            newVect.push_back(i);
        }
    }
    return IntegerSet(newVect);
}
