int quo(int x, int y){
    return x / y;

int rem(int x, int y){
    return x % y;
}
void digits(int num){
    int r{0};
    int q{0};
    while (num > 0){
        r = rem(num, 10);
        q = quo(num, 10);
        cout << r << " ";
        num = q;
    }
    cout << endl;
}
void digitsR(int num){
    int r{0};
    if (num > 0){
        r = rem(num, 10);
        digitsR(quo(num, 10));
        cout << r << " ";
    }
}

    seconds += 60 * minutes;
    seconds += 60 * 60 * hours;
    if (seconds > (12 * 60 * 60)){
        return (seconds - (12 * 60 * 60));
    }
    else{
        return seconds;
    }
}

void isPerfect(int number){
    std::stringstream outStr{};
    int sum{0};
    for (int i{1}; i < number; i++){
        if (number % i == 0){
            outStr << i << ", ";
            sum += i;
        }
    }
    if (number == sum){
        cout << "[" << outStr.str() << "]" << " " << number << " is Perfect" << endl;
    }
}
void perfectSeries(int endNumber){
    for (int i{1}; i <= endNumber; i++){
        isPerfect(i);
    }
}

int power(int base, int exp){
    if (exp == 0){
        return 1;
    }
    else{
        return base * power(base, exp - 1);
    }
}

    long result{};
    if (num >= 1){
        result = num * factRec(num - 1);
    }

        return 1;
    }

    int die2{1 + rand() %6};

    cout << "Player Rolled: " << die1 << " + " << die2 << " = " << sum << endl;

int craps(){
    enum class Status{CONTINUE, WON, LOST};
    Status gameStatus;
    unsigned int myPoint{0};
    unsigned int sumOfDice{rollDice()};
    switch (sumOfDice){
        case 7:
        case 11:
            gameStatus = Status::WON;
            break;
        case 2:
        case 3:
        case 12:
            gameStatus = Status::LOST;
            break;
        default:
            gameStatus = Status::CONTINUE;
            myPoint = sumOfDice;
            cout << "Point = " << myPoint << endl;
            break;
    }
    while (gameStatus == Status::CONTINUE){
        sumOfDice = rollDice();
        if (sumOfDice == myPoint){
            gameStatus = Status::WON;
        }
        else {
            if (sumOfDice == 7){
                gameStatus = Status::LOST;

            }
        }
    }
    if (Status::WON == gameStatus){
        return 1;
    }
    else {

        return 2;
    }
}
void gameOfCraps(int balance = 1000){
    int cont{1};
    int wager{0};
    int status{0};
    while (cont != 0 && balance > 0){
        srand(static_cast<unsigned int>(time(0)));
        status = 0;
        cout << "\nBalance: $" << balance << endl;
        cout << "Enter Your Wager: $";
        cin >> wager;
        while (wager <= 0 || wager > balance){
            cout << "Invalid Wager, Please Re-enter" << endl;
            cout << "Enter Your Wager: $";
            cin >> wager;
        }
        while (status == 0){
            status = craps();
            if (status == 1){
                balance += wager;
                cout << "You Win!" << "\nBalance: $" << balance << endl;
            }
            else{
                balance -= wager;
                cout << "You Lose :(" << "\nBalance: $" << balance << endl;
            }
        }
        if (balance == 0){
            cout << "You're broke, go home.";
            cont = 0;
        }
        else{
            cout << "Enter 0 to quit or any other value to play again: ";
            cin >> cont;
        }
    }
}
