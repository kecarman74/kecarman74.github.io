void testIntSet() {
    vector<int> arr{55, 13};
    IntegerSet one(arr);
    one.deleteElement(55);
    one.insertElement(11);
    IntegerSet two(arr);

    cout << two.toString() << endl;
    IntegerSet three{one.intersectionOfSets(two)};
    IntegerSet four{one.unionOfSets(two)};
    cout << three.toString() << endl;
    cout << four.toString() << endl;
}
