#class for dealing with fractions
class Fraction:
    #default constructor
    def __init__(self, num, den):
        self.num = num
        self.den = den
        if self.den == 0:
            raise Exception('div by zero')
        x = self.gcd()
        self.num = self.num // x
        self.den = self.den // x
        if self.den < 0:
            self.den *= -1
            self.num *= -1
    #greatest common divisor
    def gcd(self):
        b = self.num
        a = self.den
        while b != 0:
            t = b
            b = a % b
            a = t
        return a
    #to string operator overload
    def __str__(self):
        if self.den == 1:
            return str(self.num)
        else:
            return (str(self.num) + '/' + str(self.den))

    #addition of fractions operator overload
    def __add__(self, other):
        return Fraction(self.num * other.den + self.den * other.num, self.den * other.den)

    #subtraction of fractions operator overload
    def __sub__(self, other):
        return Fraction(self.num * other.den - self.den * other.num, self.den * other.den)

    #multiplication of fraction operator overload
    def __mul__(self, other):
        return Fraction(self.num * other.num, self.den * other.den)

    # division of fraction operator overload
    def __floordiv__(self, other):
        return Fraction(self.num * other.den, self.den * other.num)

    #greater than operator overload
    def __gt__(self, other):
        if (self.num / self.den) > (other.num / other.den):
            return True
        else:
            return False

    # greater than or equal to operator overload
    def __ge__(self, other):
        if (self.num / self.den) >= (other.num / other.den):
            return True
        else:
            return False

    # less than operator overload
    def __lt__(self, other):
        if (self.num / self.den) < (other.num / other.den):
            return True
        else:
            return False

    # less than or equal to than operator overload
    def __ge__(self, other):
        if (self.num / self.den) <= (other.num / other.den):
            return True
        else:
            return False
