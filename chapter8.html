void tortoiseAndHareRace() {

    string race[N];
    string* ptr{race};
    int tortoisePosition{0};
    int harePosition{0};
    cout << "Bang!" << endl;
    cout << "And They Are Off!" << endl;

    while (harePosition <= N && tortoisePosition <= N) {
        cout << endl;
        ptr = race;
        harePosition += moveHare();
        tortoisePosition += moveTortoise();

        if (tortoisePosition < 0) {
            tortoisePosition = 0;
        }

        if (harePosition < 0) {
            harePosition = 0;
        }

        for (int i{0}; i < N; i++) {

            if (i == harePosition && i == tortoisePosition) {
                *ptr = "Ouch";
            }

                else{
                    if (i == harePosition) {
                    *ptr = "H";
                }

                    else {

                            if (i == tortoisePosition) {
                            *ptr = "T";
                            }

                            else {
                            *ptr = "=";
                            }
                    }
                }
                ptr++;
        }
        printCArray(race);
    }

    if (tortoisePosition >= N) {
        cout << "\nTortoise Wins" << endl;
    }

    else {
        cout << "\nHare Wins" << endl;
    }
}

class SimpleTron{
    public:

        void simpletronEntry() {

            int input{};
            int instNum{};

            cout << "*** Welcome to Simpletron! ***" << endl;
            cout << "*** Please enter your program one instruction ***" << endl;
            cout << "*** (or data word) at a time. I will type the ***" << endl;
            cout << "*** location number and a question mark (?). ***" << endl;
            cout << "*** You then type the word for that location. ***" << endl;
            cout << "*** Type the sentinel -99999 to stop entering ***" << endl;
            cout << "*** your program. ***" << endl;

            while (instNum < 100) {
                cout << instNum << " ? ";
                cin >> input;

                if (input == -99999) {
                    break;
                }

                    else {
                            if ((input < -9999 && input != -99999) || input > 9999) {
                        cout << "Invalid Entry, please Re-enter" << endl;
                        cout << instNum << " ? ";
                        cin >> input;
                    }

                    else {
                        memory[instNum] = input;
                        instNum++;
                    }
                }
            }
            cout << "*** Program loading completed ***" << endl;
            cout << "*** Program execution begins ***" << endl;
        }

        void getCommand() {
            while (instructionNum < 100) {
                instructionReg = memory[instructionNum];
                operationCode = instructionReg / 100;
                operand = instructionReg % 100;

                switch(operationCode) {
                case 10:
                    cout << "Enter Number ";
                    cin >> memory[operand];
                    break;
                case 11:
                    cout << memory[operand] << endl;
                    break;
                case 20:
                    accumulator = memory[operand];
                    break;
                case 21:
                    memory[operand] = accumulator;
                    break;
                case 30:
                    accumulator += memory[operand];
                    break;
                case 31:
                    accumulator -= memory[operand];
                    break;
                case 32:
                    if (memory[operand] == 0) {
                        cout << "*** Attempt to divide by zero ***" << endl;
                        cout << "Simpletron Go Sleep Sleep ***" << endl;
                        instructionNum = 100;
                    }
                    else {
                        accumulator /= memory[operand];
                        break;
                    }
                case 33:
                    accumulator *= memory[operand];
                    break;
                case 40:
                    instructionNum = operand;
                    break;
                case 41:
                    if (accumulator < 0) {
                        instructionNum = operand;
                    }
                    break;
                case 42:
                    if (accumulator == 0) {
                        instructionNum = operand;
                    }
                    break;
                case 43:
                    instructionNum = 100;
            }
                instructionNum++;
            }
            memoryDump();
        }

        void memoryDump() {
            cout << "Registers:" << endl;
            cout << "Accumulator" << setw(11) << accumulator << endl;
            cout << "instructionCounter" << setw(4) << instructionNum << endl;
            cout << "instructionRegister" << setw(3) << instructionReg << endl;
            cout << "operationCode" << setw(9) << operationCode << endl;
            cout << "operand" << setw(15) << operand << endl;
            cout << "\nMEMORY:" << endl;

            for (int i{0}; i < 10; i++) {
                cout << setw(6) << i;
            }

            for (int j{0}; j < 100; j += 10 ) {
                cout << noshowpos << "\n" << j << " ";

                for (int k{0}; k < 10; k++) {
                    cout << showpos << setw(5) << setfill('0') << internal  << memory[j + k] << " ";
                }
            }
        }

        void simpletronExecute() {
            simpletronEntry();
            getCommand();
        }

    private:
        int memory[100];
        int accumulator{0};
        int instructionNum{0};
        int operationCode{0};
        int operand{0};
        int instructionReg{0};
};
