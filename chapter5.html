class DollarAmount{
public:
        explicit DollarAmount(int64_t dollars, int64_t cents) : dollarCents{100 * dollars}, allCents{cents} {}

        void add(DollarAmount right){
            amount += right.amount;
        }
        void subtract(DollarAmount right){
            amount -= right.amount;
        }
        void divide(DollarAmount right){
            amount /= right.amount;
        }
        void addInterest(int64_t rate, int64_t divisor){
            int64_t tempAmount{0};
            tempAmount = ((amount * rate + divisor / 2) / divisor);
            if (tempAmount % 1 <= 5){
                    tempAmount -= (tempAmount % 1);
            }
            amount += tempAmount;
        }
        string intRateDisplay( int rate, int precision){
            string wholeRate{rate / (precision / 100)};
            string decRate{rate % (precision / 100)};
            return wholeRate + "." + decRate + "%";
        }
        string to_String() const{
            string dollars{std::to_string(amount / 100)};
            string cents{std::to_string (std::abs(amount % 100))};
            return dollars + "." + (cents.size() == 1 ? "0" : "") + cents;
        }
private:
    int64_t dollarCents;
    int64_t allCents;
    int64_t amount{dollarCents + allCents};
};

int64_t factorial(int64_t number){
    int64_t total{1};
    int counter{1};
   while (counter != (number + 1)){
        if (counter == 1){
            cout << "1! = 1" << endl;
        }
        else {
            total = counter * total;
            cout << counter << "! = " << total << endl;
        }
        counter++;
    }
}
