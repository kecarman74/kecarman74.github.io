class Account {
    public:
        Account(std::string acctName)//constructor
        : name{acctName}, balance{0.00} {} //initializer list

        std::string getName() const { //const means changes cannot be made
            return name;
        }

        double getBalance() const {
            return balance;
        }
        void deposit(double depAmt) {
            if (depAmt > 0.00){
                balance += depAmt;
            }
        }
        void withdraw(double witAmt) {
            if (witAmt <= balance){
                balance -= witAmt;
            }
                else {
                    std::cout << "Insufficient Funds" << std::endl;
                }
        }

    private:
        std::string name{};
        double balance{0.00};

};

//3.10
class Invoice {
    public:
        Invoice(std::string partNo, std::string partDescr, int partQuant, double partPrice )//constructor
        : pn{partNo}, descr{partDescr}, quant(partQuant), price(partPrice) {} //initializer list

        std::string getPartNo() const { //const means changes cannot be made
            return pn;
        }

        std::string getPartDescr() const { //const means changes cannot be made
            return descr;
        }

        int getPartQuant() const { //const means changes cannot be made
            return quant;
        }

        double getPartPrice() const { //const means changes cannot be made
            return price;
        }

        void setPartNo(std::string newPartNo) {
            pn = newPartNo;
        }

        void setPartDescr(std::string newPartDescr) {
            descr = newPartDescr;
        }
        void setPartQuant(int newQuant) {
            if (newQuant >= 0){
                quant = newQuant;
            }
        }
        void setPartPrice(int newPrice) {
            if (newPrice >= 0){
                price = newPrice;
            }
        }
        double getInvoiceAmount() {
            if (quant < 0) {
                quant = 0;
            }
            if (price < 0) {
                price = 0;
            }
            return quant * price;

        }

    private:
        std::string pn{};
        std::string descr{};
        int quant{0};
        double price{0.00};

};

class Date {
    public:
    Date(int day, int month, int year)
    : dd{day}, mm{month}, yyyy{year} {}

    int getDay() {
        return dd;
    }

    int getMonth() {
        if ((mm < 1) || (mm > 12)) {
            mm = 1;
    }

        return  mm;
    }

    int getYear() {
        return yyyy;
    }
    void setDay(int newDay) {
        dd = newDay;
    }

    void setMonth(int newMonth) {
        mm = newMonth;
    }

    void setYear (int newYear) {
        yyyy = newYear;
    }
    private:
        int dd{};
        int mm{};
        int yyyy{};
};
