void mpgTest(){
    int trips{};
    int miles{};
    double gallons{};
    int totalMiles{};
    double totalGallons{};
    cout << "How Many trips? ";
    cin >> trips;
    while (trips > 0) {
        cout << "How Many miles? ";
        cin >> miles;
        cout << "How Many gallons? ";
        cin >> gallons;
        totalMiles += miles;
        totalGallons += gallons;
        cout << "Miles = " << miles << endl;
        cout << "Gallons = " << gallons << endl;
        cout << "Miles Per Gallon This Trip = " << miles / gallons << endl;
        cout << "Miles Per Gallon Average = " << totalMiles / totalGallons << endl;
        trips -= 1;
    }
}

int factorial(int precision){
    int sum{0};
    if (precision == 1){
        sum = 1;
    }
    else {
        sum += precision * (precision - 1);
        precision -= 1;
            while (precision > 1){
                sum = sum * (precision - 1);
                precision -= 1;
                }
            }
    return sum;
}

double eToTheX(int xTerm, int precision){
    int sum{1 + xTerm};
    while (precision >= 1 ){
        sum += pow(xTerm, (precision)) / factorial(precision);
        precision -= 1;
        }
    return sum;

}

class Account {
    public:
        Account(std::string acctName)//constructor
        : name{acctName}, balance{0.00} {} //initializer list

        std::string getName() const { //const means changes cannot be made
            return name;
        }
        double getBalance() const {
            return balance;
        }
        void deposit(double depAmt) {
            if (depAmt > 0.00){
                balance += depAmt;
            }
        }
        void withdraw(double witAmt) {
            if (witAmt <= balance){
                balance -= witAmt;
            }
            else {
                std::cout << "Insufficient Funds" << std::endl;
            }
        }
    private:
        std::string name{};
        double balance{0.00};
};

}
